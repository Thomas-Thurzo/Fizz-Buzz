                            // Start Funktion 
                            function getValues(){

                                // Get Zahlen
                                let fizzZahl = document.getElementById("fizzZahl").value;
                                let buzzZahl = document.getElementById("buzzZahl").value;

                                // Umwandeln Zahlen in Integer
                                fizzZahl = parseInt(fizzZahl);
                                buzzZahl = parseInt(buzzZahl);

                                // Check Eingabe
                                if(fizzZahl == buzzZahl){
                                    alert("Bitte zwei unterschiedliche Zahlen eingeben!");
                                }
                                else{
                                    // Zahlen anzeigen
                                anzeigeFunktion(fizzZahl, buzzZahl);
                                } 
                            }

                            // Anzeige Funktion
                            function anzeigeFunktion(fizzZahl, buzzZahl){

                                let templateRows = "";

                                for (let index = 0; index <= 100; index++) {

                                    if(index % fizzZahl == 0 && index % buzzZahl == 0 && index != 0){
                                        templateRows += `` + "FIZZ-BUZZ" + ``;
                                    }
                                    else if(index % fizzZahl == 0 && index != 0){
                                        templateRows += `` + "Fizz" + ``;
                                    }
                                    else if(index % buzzZahl == 0 && index != 0){
                                        templateRows += `` + "Buzz" + ``;
                                    }
                                    else{
                                        templateRows += `` + index + ``;
                                    }     
                                }

                                document.getElementById("ergebnis").innerHTML = templateRows;
                            }
FIZZ-BUZZ

Fizz-Buzz wurde mit zwei Funktionen geschrieben.

Die erste Funktion holt die Zahlen aus dem Eingabe Formular, rechnet sie um in Integer Werte, überprüft ob die Zahlen unterschiedlich sind und führt zuletzt die zweite Funktion aus.

Die zweite Funktion führt die Berechnungen durch und gibt die Zahlen wieder aus.
